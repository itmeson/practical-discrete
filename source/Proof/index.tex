@interact
def _(n=(1..100)):
   summation = " + ".join([str(x) for x in range(1,n+1)])
   product = n*(n+1)/2
   html('Left: $%s\;=\;%s$'%(summation,sum(range(1,n+1))))
   html('Right: $\\frac{n(n+1)}{2}\;=\;\\frac{%s(%s+1)}{2}\;=\;%s$'%(n,n,product))
